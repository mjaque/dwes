<?php
// Mostrar todos los errores
error_reporting(E_ALL);
ini_set('display_errors', 1);
ini_set('display_startup_errors', 1);

// Log de errores

// Tiempo máximo de ejecución de errores
ini_set('max_execution_time', 30);
?>
